$$
\begin{aligned}
\left(
	\begin{array}{cccc|c}
	3 & 3 & 2 & 2 & 1 \\
	2 & 2 & 1 & 1 & 0
	\end{array}
\right) 
= 
\left(
	\begin{array}{cccc|c}
	3 & 3 & 2 & 2 & 1 \\
	6 & 6 & 3 & 3 & 0
	\end{array}
\right) 
=
\left(
	\begin{array}{cccc|c}
	3 & 3 & 2 & 2 & 1 \\
	0 & 0 & -1 & -1 & -2
	\end{array}
\right)
=
\left(
	\begin{array}{cccc|c}
	3 & 3 & 0 & 0 & -3 \\
	0 & 0 & 1 & 1 & 2
	\end{array}
\right)
\\\\=
\left(
	\begin{array}{cccc|c}
	1 & 1 & 0 & 0 & -1 \\
	0 & 0 & 1 & 1 & 2
	\end{array}
\right)
\end{aligned}
$$

$$
\begin{pmatrix}
